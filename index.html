create table if not exists stockiste_completions ( id uuid primary key default gen_random_uuid(), store_id text not null, stockiste_name text not null, task_id text not null, task_name text not null, period text not null, completed_at timestamptz default now(), date date default current_date ); create table if not exists stockiste_notes ( id uuid primary key default gen_random_uuid(), store_id text not null, stockiste_name text not null, task_id text not null, task_name text not null, note text, photo_url text, date date default current_date, created_at timestamptz default now() ); create table if not exists stockiste_feedback ( id uuid primary key default gen_random_uuid(), store_id text not null, stockiste_name text not null, feedback text, photo_url text, date date default current_date, created_at timestamptz default now() );
